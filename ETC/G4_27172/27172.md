#include <bits/stdc++.h>

using namespace std;

int N;
int ls[100004];
int board[1000004];
int ans[100004];

int main() {
    cin.tie(NULL);
    cout.tie(NULL);
    ios::sync_with_stdio(false);
    cin >> N;
    for(int i = 0; i < N; i++) {
        cin >> ls[i];
    }

    for(int i = 0; i < N; i++) {
        memset(board, 0, sizeof(board));
        for (int j = ls[i] * 2; j <= 1000000; j += ls[i]) {
            board[j] = 1;
        }
        for (int j = 0; j < N; j++) {
            if(board[ls[j]] == 1) {
                ans[j] -= 1;
                ans[i] += 1;
            }
        }
    }

    for(int i = 0; i < N; i++) {
        cout << ans[i] << " ";
    }
}
